local player = game.Players.LocalPlayer
local humanoid = player.Character and player.Character:FindFirstChildOfClass("Humanoid")
local currentTrack = nil
local selectedAnim = nil

local function playAnimation(id)
    if not humanoid then humanoid = player.Character and player.Character:FindFirstChildOfClass("Humanoid") end
    if humanoid then
        local animation = Instance.new("Animation")
        animation.AnimationId = "rbxassetid://" .. id
        local track = humanoid:LoadAnimation(animation)
        track.Looped = true
        track:Play()
        return track
    end
end

local function stopAnimation()
    if currentTrack then
        currentTrack:Stop()
        currentTrack:Destroy()
        currentTrack = nil
    end
end

local animOptions = {}
for name, _ in pairs(animations) do
    table.insert(animOptions, name)
end

GamepassFe:AddDropdown({
    Name = "اختر حركة",
    Options = animOptions,
    Default = nil,
    Callback = function(Value)
        selectedAnim = animations[Value]
    end
})

GamepassFe:AddToggle({
    Name = "تشغيل/إيقاف الحركة",
    Default = false,
    Callback = function(Value)
        if Value then
            if selectedAnim then
                stopAnimation()
                currentTrack = playAnimation(selectedAnim)
            end
        else
            stopAnimation()
        end
    end
})
